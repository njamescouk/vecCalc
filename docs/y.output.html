$accept :
  definitionList $end
;
definitionList :
  definition
| definitionList definition
;
definition :
  pointDefinition
| lineDefinition
| angleDefinition
| circleDefinition
| varDefinition
| SVG_DIRECTIVE SVG_FLAG
| INC_DIRECTIVE FILENAME
;
pointDefinition :
  POINT name xcoord ycoord
| POINT name INTERSECT lineRef lineRef
| POINT name BISECT pointRef pointRef
| POINT name BISECT lineRef
| POINT name pointRef ADD pointRef
| POINT name pointRef SUBTRACT pointRef
| POINT name pointRef MULTIPLY INTEGER
| POINT name pointRef DIVIDE INTEGER
| POINT name pointRef pointRef pointRef
;
lineDefinition :
  LINE name pointRef pointRef
| LINE name PERPENDICULAR lineRef pointRef
| LINE name PARALLEL lineRef pointRef
| LINE name EXTEND lineRef INTEGER
| LINE name SLIDE lineRef INTEGER
;
circleDefinition :
  CIRCLE name pointRef pointRef pointRef
| CIRCLE name pointRef INTEGER
| CIRCLE name pointRef pointRef
| ARC name INTEGER pointRef pointRef pointRef
| ARC name INTEGER pointRef INTEGER INTEGER
;
angleDefinition :
  ANGLE name pointRef pointRef pointRef
;
varDefinition :
  VAR name INTEGER
;
xcoord :
  INTEGER
| VAR_REF
;
ycoord :
  INTEGER
| VAR_REF
;
name :
  ID
;
pointRef :
  POINT_REF
;
lineRef :
  LINE_REF
;
