$accept :
  definitionList $end
;
definitionList :
  definition
| definitionList definition
;
definition :
  pointDefinition
| lineDefinition
| SVG_DIRECTIVE SVG_FLAG
;
pointDefinition :
  POINT name xcoord ycoord
| POINT name INTERSECT lineRef lineRef
| POINT name BISECT pointRef pointRef
| POINT name BISECT lineRef
;
lineDefinition :
  LINE name pointRef pointRef
| LINE name PERPENDICULAR lineRef pointRef
| LINE name PARALLEL lineRef pointRef
| LINE name EXTEND lineRef INTEGER
| LINE name SLIDE lineRef INTEGER
| error
;
xcoord :
  INTEGER
;
ycoord :
  INTEGER
;
name :
  ID
;
pointRef :
  POINT_REF
;
lineRef :
  LINE_REF
;
